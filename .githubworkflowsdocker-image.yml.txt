name: Build and Push Docker Image

on:
  push:
    branches:
      - main

jobs:
  build-and-push:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout code
        uses: actions/checkout@v2

      - name: Login to Docker Hub
        uses: docker/login-action@v1
        with:
          username: jitinnallam
          password: Jitin@2590

      - name: Build and Push Docker Image
        run: |
          docker build -t jitinnallam/wisecow:latest .
          docker push jitinnallam/wisecow:latest

      - name: Deploy to Kubernetes
        uses: kubernetes/deploy-action@v1
        with:
          kubeconfig: ${{ secrets.KUBECONFIG }}
          deployment: wisecow-deployment
          container: wisecow
          image: jitinnallam/wisecow:latest
